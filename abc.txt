<script>
var inner = document.getElementById("lay1");
inner.addEventListener('click',innerFunction);
function innerFunction(event){
  event.stopPropagation();
  console.log("Inner Functiuon");
}
function dragStart(event) {
  event.dataTransfer.setData("Text", event.target.id);
}

function dragging(event) {
  document.getElementById("demo").innerHTML = "The p element is being dragged";
}

function allowDrop(event) {
  event.preventDefault();
}

function copyElement(elmnt){
  var div = document.createElement(elmnt.tagName);
  div.innerHTML += elmnt.outerHTML;
  return div;
}
function drop(event) {
event.preventDefault();
  var data = event.dataTransfer.getData("Text");
 event.target.appendChild(copyElement(document.getElementById(data)));
  document.getElementById("demo").innerHTML = "The p element was dropped";
}
</script>
